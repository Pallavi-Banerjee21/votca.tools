option(EXTERNAL_BOOST "Use external boost" OFF)
if (EXTERNAL_BOOST)
  find_package(Boost 1.42.0 REQUIRED COMPONENTS program_options )
  include_directories(${Boost_INCLUDE_DIRS})
  set (BOOST_LIBRARY ${Boost_PROGRAM_OPTIONS_LIBRARY})
  set (BOOST_PKG)
else(EXTERNAL_BOOST)
  include_directories(${CMAKE_SOURCE_DIR}/src/libboost)
  add_subdirectory(libboost)
  set (BOOST_LIBRARY votca_boost)
  set (BOOST_PKG "libvotca_boost")
endif(EXTERNAL_BOOST)
add_subdirectory(libtools)
